调用过程

Cgo 将Go代码编译成动态链接库，然后Java通过JNA 来调用动态链接库

1

Go ---> Cgo --> Shared libraries <--- JNA <--- Java

名词解释

Cgo：Go和C的桥梁，两者可以互相调用

JNA：一个Java框架，让Java程序更方便地使用动态链接库

package com.zzz;

import java.io.File;
import java.io.FileOutputStream;
import java.io.InputStream;

import com.sun.jna.Library;
import com.sun.jna.Native;
import com.zzz.go.types.GoString;

import org.apache.commons.io.IOUtils;

public class Demo {
    //
    public static void main(String[] argv) {

    }

    /**
     * 将指定资源从jar包里抽取到本地磁盘上
     */
    private static String extractResource(String fileName, String sourcePath) {
	File temp = new File(fileName);
	if (temp.exists()) {
	    return temp.getAbsolutePath();
	}

	FileOutputStream out = null;
	InputStream in = Demo.class.getClassLoader().getResourceAsStream(sourcePath);
	try {
	    out = new FileOutputStream(temp);
	    IOUtils.copy(in, out);
	} catch (Exception e) {
	    throw new RuntimeException("加载so 库失败!", e);
	} finally {
	    IOUtils.closeQuietly(in);
	    IOUtils.closeQuietly(out);
	}

	return temp.getAbsolutePath();
    }
}
